SELECT  count(app_profile.user_id)
FROM    (
  SELECT  user_id
  ,is_last_active_mon
  ,is_trade
  ,is_new
  FROM    appdd
  WHERE is_last_active_mon=1 and is_trade=1
) app_profile
JOIN    (
  SELECT  user_id
  ,gn_user_id
  ,register
  ,login_day_r7d
  ,sign_cancel
  FROM    basedd
  WHERE login_day_r7d>2
) ac_profile
ON      app_profile.user_id = ac_profile.user_id;
